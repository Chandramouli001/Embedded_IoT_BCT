Arduino Programming – Basics

1. Structure of Every Arduino Program

Defines the basic structure where setup runs once and loop runs repeatedly.

void setup() {
  // initialization code
}

void loop() {
  // main code runs repeatedly
}

2. Comments

Comments are used to explain code and are ignored by the compiler.

// This is a single-line comment

  /* This is a multi-line comment
used for explanation /*

3. Variables & Data Types

Variables store data values used in the program.

Syntax: dataType variableName = value;
int a = 10;
float b = 3.14;
char c = 'A';
bool d = true;

4. Constants

Constants store fixed values that cannot be changed during execution.

Syntax: const dataType variableName = value;
const int ledPin = 13;

5. pinMode()

Used to configure a pin as INPUT or OUTPUT.

Syntax: pinMode(pin, mode);
void setup() {
  pinMode(13, OUTPUT);
}

6. digitalWrite()

Sets a digital pin HIGH or LOW.

Syntax: digitalWrite(pin, value);
digitalWrite(13, HIGH);
digitalWrite(13, LOW);

7. digitalRead()

Reads the state of a digital input pin.

Syntax: digitalRead(pin);
int value = digitalRead(2);

8. analogRead()

Reads analog input values (0–1023).

Syntax: analogRead(pin);
int sensorValue = analogRead(A0);

9. analogWrite()

Outputs PWM value to control brightness or speed.

Syntax: analogWrite(pin, value);
analogWrite(9, 128);

10. delay()

Pauses program execution for a specified time.

Syntax: delay(milliseconds);
delay(1000);

11. Logical Operators

Used to combine multiple conditions.

if (a > 10 && b < 20) {
}

12. if / else

Executes code based on conditions.

if (value > 500) {
  digitalWrite(13, HIGH);
} else {
  digitalWrite(13, LOW);
}

13. for Loop

Repeats a block of code fixed number of times.

for (int i = 0; i < 5; i++) {
  digitalWrite(13, HIGH);
  delay(500);
}

14. while Loop

Repeats code while condition remains true.

while (digitalRead(2) == LOW) {
  digitalWrite(13, HIGH);
}

15. User Defined Functions

Functions help reuse code and improve readability.

returnType functionName(parameters) {

} 
void blinkLED(int pin) {
  digitalWrite(pin, HIGH);
  delay(500);
  digitalWrite(pin, LOW);
}

16. Serial Communication

Used to send and receive data with computer.

void setup() {
  Serial.begin(9600);
}

void loop() {
  Serial.println("Hello Arduino");
}

17. EEPROM

Stores data permanently even after power off.

// Include EEPROM library
#include <EEPROM.h>

EEPROM.write(0, 100);
int value = EEPROM.read(0);

Example 1: Basic LED Blink

Turns an LED ON and OFF repeatedly using pinMode and digitalWrite.

int ledPin = 13;

void setup() {
  pinMode(ledPin, OUTPUT);
}

void loop() {
  digitalWrite(ledPin, HIGH);
  delay(1000);
  digitalWrite(ledPin, LOW);
  delay(1000);
}

Example 2: LED ON using Digital Output

Turns an LED ON permanently using digitalWrite.

int ledPin = 13;

void setup() {
  pinMode(ledPin, OUTPUT);
  digitalWrite(ledPin, HIGH);
}

void loop() {
}

Example 3: Read Button State

Reads a button input using digitalRead.

int buttonPin = 2;

void setup() {
  pinMode(buttonPin, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop() {
  int state = digitalRead(buttonPin);
  Serial.println(state);
  delay(500);
}

Example 4: Button Controlled LED (if / else)

Turns LED ON or OFF based on button press using if-else.

int ledPin = 13;
int buttonPin = 2;

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(buttonPin, INPUT_PULLUP);
}

void loop() {
  if (digitalRead(buttonPin) == HIGH) {
    digitalWrite(ledPin, HIGH);
  } else {
    digitalWrite(ledPin, LOW);
  }
}

Example 5: LED Blink using for Loop

Blinks an LED multiple times using a for loop.

int ledPin = 13;

void setup() {
  pinMode(ledPin, OUTPUT);
}

void loop() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPin, HIGH);
    delay(500);
    digitalWrite(ledPin, LOW);
    delay(500);
  }
}

Example 6: Button Controlled LED Blink using Function

When the button is pressed, the LED blinks 5 times using a user-defined function and a for loop.

int ledPin = 8;
int btn = 2;

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(btn, INPUT_PULLUP);
}

void loop() {
  if (digitalRead(btn) == LOW) {
    blink();
  }
}

void blink() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPin, HIGH);
    delay(500);
    digitalWrite(ledPin, LOW);
    delay(500);
  }
}
