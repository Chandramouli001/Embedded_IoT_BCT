01 Basic Structure 
#include <REG51.H>   // header file

void main() {
    while(1) {       // infinite loop
    }
}
👉 Explanation
  • REG51.H → gives access to ports (P1, P2, etc.)
  • main() → starting point
  • while(1) → runs forever (embedded systems never stop)
02 Delay Function 
void delay() {       // delay
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}
👉 Explanation
  • Uses loops to waste time
  • Makes LED blinking visible
  • Not accurate, just for practice
03 Single LED Blink 
#include <REG51.H>

sbit LED = P1^0;   // LED pin

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    while(1) {
        LED = 0;   // ON
        delay();
        LED = 1;   // OFF
        delay();
    }
}
👉 Explanation
  • sbit selects pin P1.0
  • 0 → LED ON (active LOW)
  • 1 → LED OFF
  • Delay creates blinking
04 Multiple LED Blink 
#include <REG51.H>

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    while(1) {
        P1 = 0x00;   // all ON
        delay();
        P1 = 0xFF;   // all OFF
        delay();
    }
}
👉 Explanation
  • P1 controls 8 pins together
  • 0x00 → all pins LOW → LEDs ON
  • 0xFF → all pins HIGH → LEDs OFF
05 Control Specific Pins 
#include <REG51.H>

sbit LED1 = P1^0;
sbit LED2 = P1^1;

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    while(1) {
        LED1 = 0;   // ON
        LED2 = 1;   // OFF
        delay();

        LED1 = 1;
        LED2 = 0;
        delay();
    }
}
👉 Explanation
  • Controls pins individually
  • LED1 ON, LED2 OFF → then reverse
  • Good for learning pin-level control
06 LED Pattern 
#include <REG51.H>

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    while(1) {
        P1 = 0xAA;   // 10101010
        delay();
        P1 = 0x55;   // 01010101
        delay();
    }
}
👉 Explanation
  • Creates alternate LED pattern
  • Even LEDs ON → then odd LEDs ON
  • Uses binary pattern
07 Running LED 
#include <REG51.H>

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    while(1) {
        P1 = 0xFE; delay();
        P1 = 0xFD; delay();
        P1 = 0xFB; delay();
        P1 = 0xF7; delay();
        P1 = 0xEF; delay();
        P1 = 0xDF; delay();
        P1 = 0xBF; delay();
        P1 = 0x7F; delay();
    }
}
👉 Explanation
  • Only one LED ON at a time
  • Moves from P1.0 → P1.7
  • Creates running (chasing) effect
08 Switch → LED 
#include <REG51.H>

sbit SW  = P1^1;
sbit LED = P1^0;

void main() {
    while(1) {
        if(SW == 0)   // pressed
            LED = 0;  // ON
        else
            LED = 1;  // OFF
    }
}
👉 Explanation
  • Reads switch input
  • If pressed → LED ON
  • If not → LED OFF
09 Toggle LED using Switch 
#include <REG51.H>

sbit SW  = P1^1;
sbit LED = P1^0;

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    while(1) {
        if(SW == 0) {
            LED = ~LED;   // toggle
            delay();
        }
    }
}
👉 Explanation
  • Each press changes LED state
  • ~LED flips ON ↔ OFF
  • Delay avoids multiple triggers
10 Input to Output 
#include <REG51.H>

void main() {
    while(1) {
        P2 = P1;   // copy
    }
}
👉 Explanation
  • Reads data from Port 1
  • Sends same data to Port 2
  • Used for interfacing
11 Counter 
#include <REG51.H>

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    unsigned char i;

    while(1) {
        for(i=0;i<256;i++) {
            P1 = i;
            delay();
        }
    }
}
👉 Explanation
  • Counts from 0 to 255
  • Shows binary output on LEDs
  • Good for understanding binary
12 Buzzer 
#include <REG51.H>

sbit BUZ = P1^2;

void delay() {
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<1000;j++);
}

void main() {
    while(1) {
        BUZ = 1;   // ON
        delay();
        BUZ = 0;   // OFF
        delay();
    }
}
👉 Explanation
  • Turns buzzer ON and OFF
  • Works like LED blinking

🔥 Final Quick Revision

  • sbit → control single pin
  • P1 = value → control full port
  • 0 → ON (active LOW)
  • 1 → OFF
  • Delay → makes output visible
