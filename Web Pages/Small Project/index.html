1. Radar System

#include <Wire.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);

int trigL = 13;
int echoL = 12;
int trigR = 3;
int echoR = 2;

int buzzerL = 8;
int buzzerR = 7;

long duration;
int distanceL, distanceR;

void setup() {
  pinMode(trigL, OUTPUT);
  pinMode(echoL, INPUT);
  pinMode(trigR, OUTPUT);
  pinMode(echoR, INPUT);

  pinMode(buzzerL, OUTPUT);
  pinMode(buzzerR, OUTPUT);

  lcd.init();
  lcd.backlight();
}

int getDistance(int trig, int echo) {
  digitalWrite(trig, LOW);
  delayMicroseconds(2);
  digitalWrite(trig, HIGH);
  delayMicroseconds(10);
  digitalWrite(trig, LOW);

  duration = pulseIn(echo, HIGH);
  return duration * 0.034 / 2;
}

void loop() {
  distanceL = getDistance(trigL, echoL);
  distanceR = getDistance(trigR, echoR);

  lcd.clear();

  if (distanceL < 100) {
    digitalWrite(buzzerL, HIGH);
    lcd.setCursor(0, 0);
    lcd.print("Enemy LEFT");
  } else digitalWrite(buzzerL, LOW);

  if (distanceR < 100) {
    digitalWrite(buzzerR, HIGH);
    lcd.setCursor(0, 1);
    lcd.print("Enemy RIGHT");
  } else digitalWrite(buzzerR, LOW);

  if (distanceL >= 100 && distanceR >= 100) {
    lcd.setCursor(0, 0);
    lcd.print("No Enemy");
  }

  delay(300);
}

2. IR Home Automation

#include <IRremote.h>

int led1 = 7;
int led2 = 8;
int led3 = 9;
int led4 = 10;

bool s1=false,s2=false,s3=false,s4=false;

void setup() {
  Serial.begin(9600);
  IrReceiver.begin(2);

  pinMode(led1, OUTPUT);
  pinMode(led2, OUTPUT);
  pinMode(led3, OUTPUT);
  pinMode(led4, OUTPUT);
}

void loop() {
  if (IrReceiver.decode()) {

    unsigned long code = IrReceiver.decodedIRData.decodedRawData;
    Serial.println(code, HEX);

    if (code == 0xEF10BF00) { s1=!s1; digitalWrite(led1,s1); }
    if (code == 0xEE11BF00) { s2=!s2; digitalWrite(led2,s2); }
    if (code == 0xED12BF00) { s3=!s3; digitalWrite(led3,s3); }
    if (code == 0xEB14BF00) { s4=!s4; digitalWrite(led4,s4); }

    IrReceiver.resume();
  }
}

4. Automatic Street Light

int ldr = A0;
int pir = 8;
int led = 2;

void setup() {
  pinMode(pir, INPUT);
  pinMode(led, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int lightValue = analogRead(ldr);
  int motion = digitalRead(pir);

  if (lightValue < 300 && motion == HIGH) {
    digitalWrite(led, HIGH);
    delay(10000);
    digitalWrite(led, LOW);
  } else {
    digitalWrite(led, LOW);
  }

  delay(200);
}

5. Gas + Light Alarm

int gasSensor = A0;
int photoSensor = A1;

int led = 7;
int buzzer = 8;

void setup() {
  pinMode(led, OUTPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int gasValue = analogRead(gasSensor);
  int lightValue = analogRead(photoSensor);

  if (gasValue > 110 || lightValue > 500) {
    digitalWrite(led, HIGH);
    digitalWrite(buzzer, HIGH);
  } else {
    digitalWrite(led, LOW);
    digitalWrite(buzzer, LOW);
  }
}
