01 RGB LED – Auto Color Cycle 
#include <reg51.h>

// Define pins
sbit Red   = P1^0;
sbit Green = P1^1;
sbit Blue  = P1^2;

void delay(unsigned int time) {
    unsigned int i, j;
    for (i = 0; i < time; i++)
        for (j = 0; j < 120; j++);
}

void main() {
    while (1) {
        // --- RED ON for 1 second ---
        Red = 1; Green = 0; Blue = 0;
        delay(1000);

        // --- GREEN ON for 1 second ---
        Red = 0; Green = 1; Blue = 0;
        delay(1000);

        // --- BLUE ON for 1 second ---
        Red = 0; Green = 0; Blue = 1;
        delay(1000);
    }
}
👉 Explanation
  • sbit maps each color pin (P1.0, P1.1, P1.2)
  • Only one color is set 1 (ON) at a time; others are 0 (OFF)
  • delay(1000) keeps each color ON for ~1 second
  • Loop repeats → Red → Green → Blue → Red ...

02 RGB LED – Button Controlled 
#include <reg51.h>

// Define LED pins
sbit Red_LED   = P1^0;
sbit Green_LED = P1^1;
sbit Blue_LED  = P1^2;

// Define Button pins
sbit Btn_Red   = P2^0;
sbit Btn_Green = P2^1;
sbit Btn_Blue  = P2^2;

void main() {
    // Set Button pins as inputs
    Btn_Red = 1;
    Btn_Green = 1;
    Btn_Blue = 1;

    while (1) {
        if (Btn_Red == 0) {        // Button 1 pressed (Logic 0)
            Red_LED = 1; Green_LED = 0; Blue_LED = 0;
        }
        else if (Btn_Green == 0) { // Button 2 pressed
            Red_LED = 0; Green_LED = 1; Blue_LED = 0;
        }
        else if (Btn_Blue == 0) {  // Button 3 pressed
            Red_LED = 0; Green_LED = 0; Blue_LED = 1;
        }
        else {                      // No buttons pressed
            Red_LED = 0; Green_LED = 0; Blue_LED = 0;
        }
    }
}
👉 Explanation
  • Buttons on Port 2, LEDs on Port 1
  • Button pins set to 1 → configured as input
  • Button press reads as 0 (active LOW logic)
  • if / else if / else checks which button is pressed
  • If no button → all LEDs OFF
03 7-Segment Display – 0 to 9 Counter 
#include <reg51.h>

// 1. Define the segment pins
sbit a = P1^0;
sbit b = P1^1;
sbit c = P1^2;
sbit d = P1^3;
sbit e = P1^4;
sbit f = P1^5;
sbit g = P1^6;

void delay() {
    unsigned int i, j;
    for(i=0; i<1000; i++)
        for(j=0; j<120; j++);
}

// 2. Create functions for each number
void show_0() { a=1; b=1; c=1; d=1; e=1; f=1; g=0; }
void show_1() { a=0; b=1; c=1; d=0; e=0; f=0; g=0; }
void show_2() { a=1; b=1; c=0; d=1; e=1; f=0; g=1; }
void show_3() { a=1; b=1; c=1; d=1; e=0; f=0; g=1; }
void show_4() { a=0; b=1; c=1; d=0; e=0; f=1; g=1; }
void show_5() { a=1; b=0; c=1; d=1; e=0; f=1; g=1; }
void show_6() { a=1; b=0; c=1; d=1; e=1; f=1; g=1; }
void show_7() { a=1; b=1; c=1; d=0; e=0; f=0; g=0; }
void show_8() { a=1; b=1; c=1; d=1; e=1; f=1; g=1; }
void show_9() { a=1; b=1; c=1; d=1; e=0; f=1; g=1; }

void main() {
    while(1) {
        show_0(); delay();
        show_1(); delay();
        show_2(); delay();
        show_3(); delay();
        show_4(); delay();
        show_5(); delay();
        show_6(); delay();
        show_7(); delay();
        show_8(); delay();
        show_9(); delay();
    }
}
👉 Explanation
  • 7-segment has 7 pins: a b c d e f g → each controls one LED bar
  • Each show_X() function sets the right segments ON/OFF to display that digit
  • Example: show_0() turns all segments ON except g (middle bar)
  • delay() between each call keeps the digit visible for ~1 second
  • Loop counts 0 → 9 → 0 → 9 forever

04 7-Segment – Button Controlled (1, 2) 
#include <reg51.h>

// Segment pins
sbit a = P1^0; sbit b = P1^1; sbit c = P1^2;
sbit d = P1^3; sbit e = P1^4; sbit f = P1^5;
sbit g = P1^6;

// Button pins
sbit Btn1 = P2^0;
sbit Btn2 = P2^1;

void show_1() { a=0; b=1; c=1; d=0; e=0; f=0; g=0; }
void show_2() { a=1; b=1; c=0; d=1; e=1; f=0; g=1; }
void clear()  { a=0; b=0; c=0; d=0; e=0; f=0; g=0; }

void main() {
    Btn1 = 1;  // set as input
    Btn2 = 1;

    while(1) {
        if(Btn1 == 0)       // Button 1 pressed
            show_1();
        else if(Btn2 == 0)  // Button 2 pressed
            show_2();
        else
            clear();        // No button → blank display
    }
}
👉 Explanation
  • Two buttons on Port 2 → Btn1 and Btn2
  • Button pins set to 1 first → configured as input
  • Press Btn1 → display shows 1
  • Press Btn2 → display shows 2
  • No button pressed → clear() turns all segments OFF
05 LCD – Hello World (2 Lines) 
#include <reg51.h>

// Define LCD Pins
#define LCD_Data P2     // Data pins connected to Port 2
sbit RS = P3^0;         // Register Select
sbit RW = P3^1;         // Read/Write
sbit EN = P3^2;         // Enable

void delay(unsigned int time) {
    unsigned int i, j;
    for(i=0; i<time; i++)
        for(j=0; j<120; j++);
}

// Function to send Commands to LCD
void lcd_cmd(unsigned char cmd) {
    LCD_Data = cmd;
    RS = 0;             // 0 = Command mode
    RW = 0;
    EN = 1;             // Pulse High
    delay(1);
    EN = 0;             // Pulse Low
}

// Function to send Data (characters) to LCD
void lcd_data(unsigned char dat) {
    LCD_Data = dat;
    RS = 1;             // 1 = Data mode
    RW = 0;
    EN = 1;             // Pulse High
    delay(1);
    EN = 0;             // Pulse Low
}

void main() {
    // Initializing the LCD
    lcd_cmd(0x38);      // 2 lines, 5x7 matrix
    lcd_cmd(0x0E);      // Display ON, Cursor ON
    lcd_cmd(0x01);      // Clear Display

    lcd_cmd(0x80);      // Move to start of Line 1

    // Writing "HELLO"
    lcd_data('H'); lcd_data('E'); lcd_data('L');
    lcd_data('L'); lcd_data('O');

    lcd_cmd(0xC0);      // Move to start of Line 2

    lcd_data('W'); lcd_data('O'); lcd_data('R');
    lcd_data('L'); lcd_data('D'); lcd_data('!');

    while(1);           // Stay here
}
👉 Explanation
  • LCD_Data = P2 → 8 data pins connected to Port 2
  • RS=0 → Command mode | RS=1 → Data (character) mode
  • EN pulse (1 → 0) tells the LCD to read the data
  • 0x38 → init 2-line LCD | 0x0E → display ON | 0x01 → clear screen
  • 0x80 → cursor to Line 1 start | 0xC0 → cursor to Line 2 start
  • Each character is sent one by one using lcd_data()

06 LCD – Scrolling Text 
#include <reg51.h>

#define LCD_Data P2
sbit RS = P3^0;
sbit RW = P3^1;
sbit EN = P3^2;

void delay(unsigned int time) {
    unsigned int i, j;
    for(i=0; i<time; i++)
        for(j=0; j<120; j++);
}

void lcd_cmd(unsigned char cmd) {
    LCD_Data = cmd;
    RS = 0; RW = 0;
    EN = 1; delay(1); EN = 0;
}

void lcd_data(unsigned char dat) {
    LCD_Data = dat;
    RS = 1; RW = 0;
    EN = 1; delay(1); EN = 0;
}

void lcd_write_string(unsigned char *str) {
    while(*str) {
        lcd_data(*str++);
    }
}

void main() {
    unsigned char i;

    lcd_cmd(0x38);  // 2 lines
    lcd_cmd(0x0C);  // Display ON, Cursor OFF
    lcd_cmd(0x01);  // Clear

    lcd_cmd(0x80);  // Start at Line 1
    lcd_write_string("LCD Interfacing with 8051 Microcontroller");
    // only up to 40 characters per line

    while(1) {
        // Shift entire display left 40 times
        for(i=0; i<40; i++) {
            lcd_cmd(0x18);  // Command to shift display left
            delay(300);     // Speed of scrolling (higher = slower)
        }
        // Return to home position to restart
        lcd_cmd(0x02);
        delay(300);
    }
}
👉 Explanation
  • lcd_write_string() → sends a full string character by character using a pointer
  • 0x0C → Display ON but cursor hidden (cleaner look for scrolling)
  • LCD can hold up to 40 characters per line (only 16 visible at once)
  • 0x18 → shifts the entire display view one step to the left
  • Loop shifts 40 times → full text scrolls across the screen
  • 0x02 → returns cursor to home (restarts scroll)
  • Increase delay(300) value to slow down scrolling

07 DS18B20 Temperature Sensor (Serial Output) 
#include <reg51.h>

sbit DQ = P2^0;   // Sensor data pin

// --- SERIAL SETUP ---
void Serial_Init() {
    TMOD = 0x20; TH1 = 0xFD; SCON = 0x50; TR1 = 1;
}

void Send_Char(char c) {
    SBUF = c; while(!TI); TI = 0;
}

void Send_String(char *s) {
    while(*s) Send_Char(*s++);
}

void Send_Temp(unsigned char t) {
    Send_Char((t/10)+'0');
    Send_Char((t%10)+'0');
    Send_String(" C\r\n");
}

// --- DELAY ---
void delay_us(unsigned int d) { while(d--); }

// --- DS18B20 FUNCTIONS ---
void ds_init() {
    DQ=0; delay_us(50);
    DQ=1; delay_us(50);
}

void ds_write(unsigned char d) {
    unsigned char i;
    for(i=0; i<8; i++) {
        DQ=0; DQ=d&1; delay_us(5);
        DQ=1; d>>=1;
    }
}

unsigned char ds_read() {
    unsigned char i, val=0;
    for(i=0; i<8; i++) {
        DQ=0; DQ=1;
        if(DQ) val |= (1<<i);
        delay_us(5);
    }
    return val;
}

// --- MAIN ---
void main() {
    unsigned char low, high, temp;
    Serial_Init();

    while(1) {
        ds_init();
        ds_write(0xCC); ds_write(0x44);  // Start conversion

        { unsigned int i; for(i=0;i<50000;i++); } // Wait for conversion

        ds_init();
        ds_write(0xCC); ds_write(0xBE);  // Read scratchpad

        low  = ds_read();
        high = ds_read();

        temp = (high<<4) | (low>>4);   // Extract °C

        Send_String("Temp: ");
        Send_Temp(temp);
    }
}
👉 Explanation
  • DQ = P2^0 → DS18B20 uses a single data wire (1-Wire protocol)
  • Serial_Init() → sets up UART at 9600 baud using Timer 1
  • ds_init() → resets the sensor by pulling DQ low then releasing
  • 0xCC → Skip ROM command (talk to all sensors on bus)
  • 0x44 → Start temperature conversion command
  • 0xBE → Read scratchpad (get temperature bytes)
  • Two bytes are read: low and high → combined to get temperature in °C
  • Send_Temp() → converts number to ASCII digits and sends via serial port
  • Output visible on any Serial Monitor (e.g. Keil, Proteus terminal)

🔥 Quick Revision – Intermediate Concepts

  • RGB = combine R, G, B pins for mixed colors
  • Button input → set pin to 1 first, read 0 when pressed
  • 7-segment → a–g pins, each digit is a fixed combination
  • RS=0 → LCD command | RS=1 → LCD data
  • 0x80 → LCD Line 1 | 0xC0 → LCD Line 2
  • 0x18 → shift LCD display left (scroll)
  • DS18B20 → 1-Wire sensor, needs init + write + read sequence
  • 0xCC, 0x44 → start temp | 0xCC, 0xBE → read temp
